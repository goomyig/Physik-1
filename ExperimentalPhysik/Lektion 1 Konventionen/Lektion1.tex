\section*{Konventionen}

\begin{enumerate}

\item \textbf{Variablennamen (mit Einheiten):}

\begin{align*}
x, y, z &:\ \text{Ortskoordinaten} \\[4pt]
t &:\ \text{Zeit} \\[4pt]
\vec{v} &:\ \text{Geschwindigkeit } [\mathrm{m/s}] \\[4pt]
a &:\ \text{Beschleunigung } [\mathrm{m/s^2}] \\[4pt]
g &:\ \text{Schwerebeschleunigung } [\mathrm{m/s^2}] \\[4pt]
F &:\ \text{Kraft } [\mathrm{N}] \\[4pt]
m &:\ \text{Masse } [\mathrm{kg}] \\[4pt]
p &:\ \text{Impuls } [\mathrm{kg\,m/s}] \\[4pt]
L &:\ \text{Drehimpuls } [\mathrm{kg\,m^2/s}] \\[4pt]
I &:\ \text{Trägheitsmoment } [\mathrm{kg\,m^2}] \\[4pt]
M &:\ \text{Drehmoment } [\mathrm{N\,m}] \\[4pt]
T &:\ \text{Seilspannung } [\mathrm{N}] \\[4pt]
P &:\ \text{Periode } [\mathrm{s}]
\end{align*}

\item \textbf{Vektorielle Größen:}

\[
\vec{v}, \vec{a}, \vec{g}, \vec{F}, \vec{p}, \vec{L}, \vec{M}, \vec{T}
\]

$p$ ist aber NICHT Druck, nur Impuls

\[
\vec{r}:\ \text{Ortsvektor}, \quad 
\vec{r} =
\begin{pmatrix}
x \\ y \\ z
\end{pmatrix}
\]

\[
\vec{r} = x\,\vec{e}_x + y\,\vec{e}_y + z\,\vec{e}_z
\]

\item \textbf{Griechische Buchstaben (ohne Einheiten):}

\[
\alpha, \beta, \gamma, \delta, \epsilon, \zeta, \eta, \theta(t), \iota, \kappa, \lambda, \mu, \nu, \xi, o, \pi, \rho, \sigma,
\]
\[
\tau, \upsilon, \phi, \chi, \psi, \omega
\]

\item \textbf{Durchgängig SI-Einheiten:} 

Basiseinheiten:
\[
\mathrm{m} \ (\text{Länge}), \quad 
\mathrm{kg} \ (\text{Masse}), \quad 
\mathrm{s} \ (\text{Zeit}), \quad 
\mathrm{A} \ (\text{Strom}), \quad 
\mathrm{K} \ (\text{Temperatur}), \quad 
\mathrm{mol}, \quad 
\mathrm{cd}
\]

Alle Größen werden daraus abgeleitet, z.\,B.:
\[
\mathrm{N} = \mathrm{kg\,m/s^2}, \quad
\mathrm{J} = \mathrm{kg\,m^2/s^2}
\]

\item \textbf{Vorsilben:} 

\[
\begin{array}{ll}
10^{-3} & \text{milli (m)} \\
10^{-6} & \text{mikro } (\mu) \\
10^{-9} & \text{nano (n)} \\
10^{3}  & \text{kilo (k)} \\
10^{6}  & \text{Mega (M)} \\
10^{9}  & \text{Giga (G)}
\end{array}
\]

\end{enumerate}
